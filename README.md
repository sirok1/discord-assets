# Example
```js
const Discord = require("discord.js");
const ASSETS = require("discord-assets");
const asset = new ASSETS();
const image = asset.sad();
const embed = new Discord.MessageEmbed()
    .setTitle(`Sad gif`)
    .setColor("GREEN")
    .setImage(image);
message.channel.send(embed);
```
